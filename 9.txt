; Приклад використання map в Лісп
(defun multiply-by-two (x) (* x 2))

(print "Output list:")
(print(map 'list #'multiply-by-two '(1 2 3 4 5)))


; Приклад використання remove-if-not в Лісп з консольним виводом
(defun is-even (x) (= (mod x 2) 0))

(let ((result (remove-if-not #'is-even '(1 2 3 4 5 6))))
  (format t "~%Output list: ~a~%" result))
